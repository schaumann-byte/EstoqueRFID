/*
  esp32_rfid_portal.ino
  - Captive portal (AP) para receber SSID+PASS via celular
  - Salva credenciais em Preferences
  - Quando conectado em STA, serve UI para controlar o Nano via Serial2
  - UI tem botões para comandos e aba "Logs" (polling /log)
  - Usa DNSServer para "captive" durante o AP (redireciona todo host para 192.168.4.1)
  
  Requisitos: Arduino core for ESP32, Bibliotecas padrão WiFi, WebServer, DNSServer, Preferences.
*/

#include <WiFi.h>
#include <WebServer.h>
#include <DNSServer.h>
#include <Preferences.h>
#include <ESPmDNS.h>
#include <HTTPClient.h>

// Serial2 pins (ajuste se necessário)
static const int PIN_RX2 = 16; // ESP32 RX2  <- Nano TX (use divisor no lado Nano)
static const int PIN_TX2 = 17; // ESP32 TX2  -> Nano RX

HardwareSerial Link(2);

// Captive DNS
const byte DNS_PORT = 53;
DNSServer dnsServer;

// Web server
WebServer server(80);

// Preferences (para armazenar ssid/senha)
Preferences prefs;
const char *PREF_NAMESPACE = "wifi";
const char *KEY_SSID = "ssid";
const char *KEY_PASS = "pass";

// AP config while provisioning
const char* AP_SSID = "ESP32-Config";
const char* AP_PASS = "config2025"; // troque se quiser

// State & log buffer
String logBuf = "";
const size_t LOG_MAX = 16*1024; // buffer de logs
bool isAP = false;

const char* API_BASE = "http://10.210.141.170:8000";  
const char* API_HEARTBEAT = "/rfid/heartbeat";

// Helper para append no log (corta quando cresce demais)
// AGORA: também imprime tudo no Serial Monitor
void appendLog(const String &s) {
  logBuf += s;

  // Joga tudo também na serial
  Serial.print(s);

  if (logBuf.length() > LOG_MAX) {
    // manter apenas final
    logBuf = logBuf.substring(logBuf.length() - LOG_MAX);
  }
}

// Id de origem (útil p/ saber de qual leitor veio)
String origemId() {
  uint64_t chipid = ESP.getEfuseMac();
  char buf[32];
  snprintf(buf, sizeof(buf), "esp32-%04X", (uint16_t)(chipid & 0xFFFF));
  return String(buf);
}

bool postHeartbeat(const String& uid, const String& origem) {
  if (WiFi.status() != WL_CONNECTED) {
    appendLog("HTTP skip (WiFi OFF)\n");
    return false;
  }
  HTTPClient http;
  String url = String(API_BASE) + String(API_HEARTBEAT);
  http.begin(url);
  http.addHeader("Content-Type", "application/json");

  // corpo JSON
  String body = "{\"etiqueta_rfid\":\"" + uid + "\",\"origem\":\"" + origem + "\"}";
  int code = http.POST(body);

  // logs
  if (code > 0) {
    appendLog("POST " + url + " [" + String(code) + "] uid=" + uid + "\n");
  } else {
    appendLog("POST fail: " + String(http.errorToString(code).c_str()) + "\n");
  }

  http.end();
  return code >= 200 && code < 300;
}

// Envia linha ao Nano e registra no log
void sendToNano(const String &line) {
  Link.print(line);
  if (!line.endsWith("\n")) Link.print("\n");
  appendLog("> " + line + "\n");
}

// ---------- Handlers ----------

// Root: if AP -> show config page, else control page
void handleRoot() {
  if (isAP) {
    server.send(200, "text/html", CONFIG_PAGE);
  } else {
    server.send(200, "text/html", CONTROL_PAGE);
  }
}

// Save credentials from form (POST /save)
void handleSave() {
  if (!server.hasArg("ssid")) {
    server.send(400, "text/plain", "ssid required");
    return;
  }
  String ssid = server.arg("ssid");
  String pass = server.arg("pass");
  ssid.trim();
  pass.trim();
  // store in preferences
  prefs.begin(PREF_NAMESPACE, false);
  prefs.putString(KEY_SSID, ssid);
  prefs.putString(KEY_PASS, pass);
  prefs.end();
  server.send(200, "text/html", "<html><body><h3>Salvo. Tentando conectar... aguarde e reinicie.</h3></body></html>");
  appendLog("Credenciais salvas: " + ssid + "\n");
  delay(500);
  // tente conectar de imediato (async): reiniciar para simplificar fluxo
  ESP.restart();
}

// POST /cmd -> envia comando para nano
void handleCmd() {
  String body = server.arg(0);
  body.trim();
  if (body.length() == 0) {
    server.send(400, "text/plain", "empty body");
    return;
  }
  sendToNano(body);
  server.send(200, "text/plain", "sent");
}

// GET /log -> retorna buffer de log
void handleLog() {
  server.send(200, "text/plain", logBuf);
}

// Fallback (serve control page static assets if any) - not used
void handleNotFound() {
  server.send(404, "text/plain", "Not found");
}

// DNS redirect (captive) - redirect everything to AP IP while in AP mode
void startCaptive() {
  // start DNS - redirect all hostnames to 192.168.4.1
  dnsServer.start(DNS_PORT, "*", WiFi.softAPIP());
}

// Connect WiFi as STA using stored creds. returns true if success.
bool connectSavedWifi() {
  prefs.begin(PREF_NAMESPACE, true); // read-only
  String ssid = prefs.getString(KEY_SSID, "");
  String pass = prefs.getString(KEY_PASS, "");
  prefs.end();

  if (ssid.length() == 0) {
    appendLog("No saved SSID\n");
    return false;
  }

  appendLog("Trying WiFi: " + ssid + "\n");
  WiFi.mode(WIFI_STA);
  WiFi.begin(ssid.c_str(), pass.c_str());

  unsigned long start = millis();
  while (WiFi.status() != WL_CONNECTED && millis() - start < 20000) {
    delay(500);
  }
  if (WiFi.status() == WL_CONNECTED) {
    IPAddress ip = WiFi.localIP();
    appendLog("WiFi connected: " + ip.toString() + "\n");
    Serial.print("IP STA: ");
    Serial.println(ip);
    return true;
  } else {
    appendLog("WiFi connect FAILED\n");
    return false;
  }
}

// Start AP mode and captive portal
void startAP() {
  WiFi.mode(WIFI_AP);
  WiFi.softAP(AP_SSID, AP_PASS);
  delay(200);
  IPAddress apIP = WiFi.softAPIP(); // normalmente 192.168.4.1
  appendLog("AP started: " + apIP.toString() + " SSID:" + String(AP_SSID) + "\n");

  Serial.println("======== MODO AP ========");
  Serial.print("SSID: ");
  Serial.println(AP_SSID);
  Serial.print("PASS: ");
  Serial.println(AP_PASS);
  Serial.print("IP AP: ");
  Serial.println(apIP);
  Serial.println("=========================");

  startCaptive();
  isAP = true;
}

void setup() {
  Serial.begin(115200);
  delay(200);
  appendLog("ESP32 boot\n");

  // Serial2 for Nano
  Link.begin(57600, SERIAL_8N1, PIN_RX2, PIN_TX2);

  // Try to connect saved wifi
  bool ok = connectSavedWifi();
  if (!ok) {
    // start captive AP
    startAP();
  } else {
    isAP = false;
  }

  // Start mDNS when in STA (useful)
  if (!isAP && MDNS.begin("esp32")) {
    appendLog("mDNS started: esp32.local\n");
  }

  // Setup web routes
  server.on("/", HTTP_GET, handleRoot);
  server.on("/save", HTTP_POST, handleSave);
  server.on("/cmd", HTTP_POST, handleCmd);
  server.on("/log", HTTP_GET, handleLog);
  server.onNotFound(handleNotFound);
  server.begin();
  appendLog("HTTP server started\n");
}

void loop() {
  // If AP mode, process DNS server
  if (isAP) dnsServer.processNextRequest();

  server.handleClient();

  while (Link.available()) {
    static String line;
    char c = Link.read();
    if (c == '\n') {
      appendLog(line + "\n");

      // Parse simples: tipo,uid,resto
      String s = line;
      s.trim();
      if (s.length()) {
        int p1 = s.indexOf(',');
        String kind = (p1 >= 0) ? s.substring(0, p1) : s;
        String uid = "";

        if (p1 >= 0) {
          int p2 = s.indexOf(',', p1 + 1);
          if (p2 < 0) p2 = s.length();
          uid = s.substring(p1 + 1, p2);
          uid.trim();
          uid.toUpperCase();
        }

        // Dispare heartbeat para tipos de interesse
        if (kind == "READ" || kind == "FOUND" || kind == "SEEN") {
          postHeartbeat(uid, origemId());
        }
      }

      line = "";
    } else if (c != '\r') {
      line += c;
    }
  }

  // Optional: if STA and WiFi disconnected, try reconnect
  if (!isAP && WiFi.status() != WL_CONNECTED) {
    appendLog("WiFi disconnected, trying reconnect\n");
    if (!connectSavedWifi()) {
      // fallback to AP
      appendLog("Switching to AP mode\n");
      startAP();
    } else {
      // restart server and mDNS
      if (MDNS.begin("esp32")) appendLog("mDNS restarted\n");
    }
  }
}
